    // define a custom font
    @font-face {
      font-family: myFont;
      src: url(sansation_light.woff);
    }
    
    // set different font-weight
    @font-face {
      font-family: myFont;
      src: url(sansation_bold.woff);
      font-weight: bold;
    }
    
    // Use in CSS
    body {
      font-family: myFont, sans-serif;
    }
    h1 {
      font-weight: bold;
    }
